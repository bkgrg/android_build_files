### Common Terminal Commands##################################################

#1. Swapfile 24GB & swappiness=80
sudo swapoff -a -v
sudo rm /swapfile
sudo dd if=/dev/zero of=/swapfile bs=1k count=25165824
mkswap /swapfile
chmod 0600 /swapfile
swapon /swapfile
echo "/swapfile swap swap auto 0 0" | tee -a /etc/fstab
sysctl -w vm.swappiness=80
echo vm.swappiness = 80 | tee -a /etc/sysctl.conf

sudo apt autoremove --purge

#3. Install commonly used packages
sudo apt-get install gedit

wget https://release.gitkraken.com/linux/gitkraken-amd64.deb && sudo apt-get install gitkraken-amd64.deb

#4. Reinstall required packages
apt list --installed
sudo apt remove xxxx
sudo apt autoremove -y
sudo rm -r /usr/bin/xxxx

#5. Find string in a folder containing files
grep -Rnw '/path/to/somewhere/' -e 'pattern'

#6. Free-up RAM
free -h && sudo sysctl -w vm.drop_caches=3 && sudo sync && echo 3 | sudo tee /proc/sys/vm/drop_caches && free -h
