#1. Swapfile 16GB & swappiness=100 & cache_pressure=10
sudo su
swapoff -a
swapoff -a -v
rm /swapfile
dd if=/dev/zero of=/swapfile bs=1M count=16000
mkswap /swapfile && chmod 600 /swapfile && swapon /swapfile
echo "/swapfile swap swap auto 0 0" | tee -a /etc/fstab
sysctl -w vm.swappiness=100
sysctl -w vm.vfs_cache_pressure=10
echo vm.swappiness=100 | tee -a /etc/sysctl.conf
echo vm.vfs_cache_pressure=10 | tee -a /etc/sysctl.conf
